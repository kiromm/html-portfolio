/* Parent and child relationship */
/* child selector */
.box > p {
    color: red;
} 
/* Above means "box" is parent, "p" is child. Must only have 1 level underneath */
/* Direcly below the div with class box is a <p> element which is the child. Anything below the <p> element are grandchild
*/


/* Ancestor and descendant relationship */
/* Descendant selector */
.box p {
    color: red;
}

/* Ancestor and Descendant relationship
This means the "box" is the ancestor and "p" is descendent
anything below the class "box" with the "p" element, change the text to color red
*/


/* Ancestor and descendant relationship */
/* Descendant selector */
.box li {
    color: blue;
}
/* This changes all list elements to blue */



/* Chaining selector */
li.done {
    color: green;
}
